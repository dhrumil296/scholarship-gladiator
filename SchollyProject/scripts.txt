create table INSTITUTE(
inst_code varchar2(50), //pk
inst_name varchar2(50),
inst_type varchar2(50),
board_name varchar2(50),
inst_pwd varchar2(50),
proof_of_existence varchar2(50),
affiliated_to varchar2(50),
inst_state varchar2(50),
inst_district varchar2(50),
inst_addr varchar2(50),
princi_name varchar2(50),
phone_no varchar2(20),
inst_status varchar2(5)
);


create table Student(
stud_id varchar2(50), //pk	
stud_name varchar2(50),
dob	varchar2(20),
gender varchar2(5),
mob_no varchar2(11),	
stud_email_id varchar2(50),
stud_pwd varchar2(50),
state_of_domicile varchar2(30),	
district varchar2(30),
bank_acc_no varchar2(14),	
bank_ifsc varchar2(10),	
inst_code varchar2(50) //fk
);

alter table student add constraint pk_02 primary key(stud_id) 

alter table institute add constraint pk_03 primary key(inst_code) 

alter table student add constraint fk_02 foreign key (inst_code) references Institute(inst_code) 


create table student_login(
stud_id varchar2(50),
stud_pwd varchar2(50)
);

alter table student_login add constraint fk1 foreign key (stud_id) references Student(stud_id) on delete cascade

create table INSTITUTE(
inst_code number(10), 
inst_name varchar2(50),
inst_type varchar2(50),
board_name varchar2(50),
inst_pwd varchar2(50),
proof_of_existence varchar2(50),
affiliated_to varchar2(50),
inst_state varchar2(50),
inst_district varchar2(50),
inst_addr varchar2(50),
princi_name varchar2(50),
phone_no varchar2(20),
inst_status varchar2(5)
);


create table Student(
stud_id number(12), 	
stud_name varchar2(50),
dob varchar2(20),
gender varchar2(5),
mob_no varchar2(11),	
stud_email_id varchar2(50),
stud_pwd varchar2(50),
state_of_domicile varchar2(30),	
district varchar2(30),
bank_acc_no varchar2(14),	
bank_ifsc varchar2(10),	
inst_code number(10)
);

alter table student add constraint STUDENT_PK PRIMARY KEY (STUD_ID)

alter table institute add constraint INSTITUTE_PK PRIMARY KEY (INST_CODE)

alter table student add constraint STUDENT_INST_FK FOREIGN KEY (INST_CODE) REFERENCES INSTITUTE(INST_CODE) ON DELETE CASCADE

create table SCHEME(			
SCHEME_ID number(12),
SCHEME_NAME varchar2(50),
SCHEME_DESCR varchar2(200),
SCHEME_AMT number(7,2),
CONSTRAINT SCHEME_PK PRIMARY KEY (SCHEME_ID)
);

create table SCHOLARSHIP_FORM(	
SCH_FORM_ID SCHOLARSHIPFORM_PK PRIMARY KEY,	
FORM_ID	number CONSTRAINT SCHOLARSHIPFORM_FORM_FK REFERENCES FORM(FORM_ID) ON DELETE CASCADE,
STUDENT_ID number CONSTRAINT SCHOLARSHIPFORM_STUDENT_FK REFERENCES STUDENT(STUD_ID) ON DELETE CASCADE,
SCHEME_ID number CONSTRAINT SCHOLARSHIPFORM_SCHEME REFERENCES SCHEME(SCHEME_ID) ON DELETE CASCADE,
FORM_STATUS int DEFAULT 0
);

create table FORM(										
FORM_ID number(12) FORM_PK PRIMARY KEY,
STUDENT_ID number CONSTRAINT FORM_STUDENT_FK REFERENCES STUDENT(STUD_ID) ON DELETE CASCADE,
SCHEME_ID number CONSTRAINT FORM_SCHEME REFERENCES SCHEME(SCHEME_ID) ON DELETE CASCADE,
RELIGION varchar2(15),
COMMUNITY varchar2(30),
FATHER_NAME varchar2(20),
MOTHER_NAME varchar2(20),
FAMILY_INCOME number(10,2),
ISDISABILITY int DEFAULT 0,
DISABILITY_TYPE varchar2(20),
PROF_FATHER varchar2(25),
PROF_MOTHER varchar2(25),
PASSYR_10 number(4),
PERC_10 number(5,2),
PASSYR_12 number(4),
PERC_12 number(5,2),
INST_NAME varchar2(35),
CURR_COURSE varchar2(25),
START_DATE date,
UNIV_NAME varchar2(35),
TOTAL_FEES number(10,2)
);
